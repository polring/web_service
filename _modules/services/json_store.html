"""JSON 데이터 디렉터리 접근 — 블루프린트에서 공통 사용.

``app/data`` 아래의 JSON 파일을 읽고, API/페이지에서 재사용합니다.
파일명 상수는 매직 문자열 분산을 줄이기 위해 모듈 수준에서 정의합니다.
"""
from __future__ import annotations

from typing import Any

_DATA_ROOT = Path(__file__).resolve().parents[1]
DATA_DIR = _DATA_ROOT / "data"

SKILLS_JSON = "skills.json"

def load_json(filename: str) -> Any:
    """``DATA_DIR`` 에서 JSON 파일을 읽어 파이썬 객체로 반환합니다.

    Args:
        filename: ``DATA_DIR`` 기준 상대 파일명 (예: ``\"profile.json\"``).

    Returns:
        ``json.load`` 결과 (보통 ``dict`` 또는 ``list``).

    Raises:
        FileNotFoundError: 파일이 없을 때.
        json.JSONDecodeError: JSON 형식이 잘못되었을 때.
    """
    path = DATA_DIR / filename
    with path.open(encoding="utf-8") as f:
        return json.load(f)

def find_item_by_id(items: list, item_id: str) -> dict | None:
    """리스트 항목에서 ``id`` 필드가 ``item_id`` 와 일치하는 dict 하나를 찾습니다.

    ``id`` 는 문자열/숫자 혼용을 허용하기 위해 비교 시 ``str`` 로 맞춥니다.

    Args:
        items: ``{\"id\": ...}`` 형태의 dict 로 이루어진 목록.
        item_id: 경로 등에서 넘어온 식별자.

    Returns:
        일치 항목이 있으면 해당 ``dict``, 없으면 ``None``.
    """
    for item in items:
        if str(item.get("id")) == str(item_id):
            return item
    return None
