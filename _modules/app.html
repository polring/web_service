"""Flask 애플리케이션 진입점 및 팩토리.

``create_app()`` 으로 앱 인스턴스를 만들고, ``main`` / ``api`` 블루프린트와
선택 플러그인, Flasgger(Swagger UI)를 등록합니다.
"""
from __future__ import annotations

import importlib
import os
import pkgutil

from flask import Flask, redirect

from blueprints.main import bp as main_bp
from blueprints.api import bp as api_bp


def _init_flasgger(app: Flask) -> None:
    """Swagger UI(``/apidocs/``)와 OpenAPI 스펙 JSON을 연결합니다.

    * 환경 변수 ``FLASGGER_DISABLE=1`` (또는 ``true``/``yes``) 이면 비활성화.
    * ``flasgger`` 패키지가 없으면 경고 로그만 남기고 건너뜀.

    Args:
        app: Flask 애플리케이션 인스턴스.
    """
    if os.getenv("FLASGGER_DISABLE", "").lower() in ("1", "true", "yes"):
        app.logger.warning("FLASGGER_DISABLE 로 Swagger UI(/apidocs)가 비활성화되었습니다.")
        return
    try:
        from flasgger import Swagger
    except ImportError:
        app.logger.warning(
            "Flasgger가 설치되어 있지 않아 /apidocs 가 제공되지 않습니다. "
            "`pip install -r app/requirements.txt` 로 설치하세요."
        )
        return

    template = {
        "swagger": "2.0",
        "info": {
            "title": "Flask Blog API",
            "description": "포트폴리오 JSON 조회 API — Flasgger + YAML(OpenAPI)",
            "version": "1.0.0",
        },
        "basePath": "/",
        "tags": [
            {"name": "Profile", "description": "프로필 단일 리소스"},
            {"name": "Projects", "description": "프로젝트 목록 및 단건"},
            {"name": "Notes", "description": "노트 목록 및 단건"},
        ],
    }
    Swagger(app, template=template)

    @app.get("/apidocs")
    def _swagger_ui_redirect():
        return redirect("/apidocs/", code=308)

    app.logger.info("Flasgger 활성화: Swagger UI → /apidocs/ (스펙 JSON → /apispec_1.json)")

def load_plugins(app: Flask) -> None:
    """``app/plugins`` 패키지 아래 모듈을 동적으로 로드합니다.

    각 모듈에 ``register(app)`` 이 있으면 호출하고, 없고 ``bp`` 가 있으면
    블루프린트로 등록합니다. ``PLUGINS=a,b`` 로 로드할 모듈 이름을 제한할 수 있습니다.

    Args:
        app: Flask 애플리케이션 인스턴스.
    """
    package_name = "plugins"
    plugins_dir = os.path.join(os.path.dirname(__file__), package_name)
    if not os.path.isdir(plugins_dir):
        return

    requested = os.getenv("PLUGINS")
    if requested:
        module_names = [name.strip() for name in requested.split(",") if name.strip()]
    else:
        module_names = [module.name for module in pkgutil.iter_modules([plugins_dir])]

    for name in module_names:
        module = importlib.import_module(f"{package_name}.{name}")
        if hasattr(module, "register"):
            module.register(app)
        elif hasattr(module, "bp"):
            app.register_blueprint(module.bp)

def create_app() -> Flask:
    """Flask 앱을 생성하고 확장(블루프린트·플러그인·Flasgger)을 등록합니다.

    Returns:
        설정이 끝난 ``Flask`` 인스턴스.
    """
    app = Flask(__name__)
    app.register_blueprint(main_bp)
    app.register_blueprint(api_bp)

    load_plugins(app)
    _init_flasgger(app)
    return app




app = create_app()


if __name__ == "__main__":
    port = int(os.environ.get("PORT", "5000"))
    app.run(debug=True, host="127.0.0.1", port=port)
